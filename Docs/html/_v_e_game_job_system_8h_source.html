232  void push(JOB* job) {
233  while (m_lock.test_and_set(std::memory_order::acquire)); // acquire lock
234 
235  job->m_next = nullptr; //clear pointer to successor
236  if (m_head == nullptr) { //if queue is empty
237  m_head = job; //let m_head point to the job
238  }
239  if (m_tail == nullptr) { //if queue was empty
240  m_tail = job; //let m_tail point to the job
241  }
242  else {
243  m_tail->m_next = (JOB*)job; //add the job to the queue tail
244  m_tail = job; //m_tail points to the new job
245  }
246 
247  m_size++; //increase size
248  m_lock.clear(std::memory_order::release); //release lock
249  };

255  JOB* pop() {
256  if (m_head == nullptr) return nullptr;
257 
258  while (m_lock.test_and_set(std::memory_order::acquire)); // acquire lock
259 
260  JOB* head = m_head;
261  if (head != nullptr) { //if there is a job at the head of the queue
262  m_head = (JOB*)head->m_next; //let point m_head to its successor
263  m_size--; //decrease number of jobs
264  if (head == m_tail) { //if this is the only job
265  m_tail = nullptr; //let m_tail point to nullptr
266  }
267  }
268  m_lock.clear(std::memory_order::release); //release lock
269 
270  return head;
271  };

446  void thread_task(int32_t threadIndex = 0) noexcept {
447  constexpr uint32_t NOOP = 10; //number of empty loops until garbage collection
448  m_thread_index = threadIndex; //Remember your own thread index number
449  static std::atomic<uint32_t> thread_counter = m_thread_count.load(); //Counted down when started
450 
451  thread_counter--; //count down
452  while (thread_counter.load() > 0) {} //Continue only if all threads are running
453 
454  uint32_t next = rand() % m_thread_count; //initialize at random position for stealing
455  thread_local uint32_t noop = NOOP; //number of empty loops until threads sleeps
456  while (!m_terminate) { //Run until the job system is terminated
457  m_current_job = m_local_queues[m_thread_index].pop(); //try get a job from the local queue
458  if (m_current_job == nullptr) {
459  m_current_job = m_global_queues[m_thread_index].pop(); //try get a job from the global queue
460  }
461  if (m_current_job == nullptr) { //try steal job from another thread
462  if (++next >= m_thread_count) next = 0;
463  m_current_job = m_global_queues[next].pop();
464  }
465 
466  if (m_current_job != nullptr) {
467  std::chrono::high_resolution_clock::time_point t1, t2;
468 
469  if (is_logging()) {
470  t1 = std::chrono::high_resolution_clock::now(); //time of finishing;
471  }
472 
473  auto is_function = m_current_job->is_function(); //save certain info since a coro might be destroyed
474  auto type = m_current_job->m_type;
475  auto id = m_current_job->m_id;
476 
477  (*m_current_job)(); //if any job found execute it - a coro might be destroyed here!
478 
479  if (is_logging()) {
480  t2 = std::chrono::high_resolution_clock::now(); //time of finishing
481  log_data(t1, t2, m_thread_index, false, type, id );
482  }
483 
484  if (is_function) {
485  child_finished((Job*)m_current_job); //a job always finishes itself, a coro will deal with this itself
486  }
487  }
488  --noop;
489  if (noop == 0) { //if none found too longs let thread sleep
490  noop = NOOP;
491  if (m_thread_index == 0) { //thread 0 is the garbage collector
492  m_delete.clear(); //delete jobs to reclaim memory
493  }
494  }
495  };

499  m_global_queues[m_thread_index].clear(); //clear your global queue
500  m_local_queues[m_thread_index].clear(); //clear your local queue
501 
502  uint32_t num = m_thread_count.fetch_sub(1); //last thread clears recycle and garbage queues
