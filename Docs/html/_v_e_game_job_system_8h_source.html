237  void push(JOB* job) {
238  while (m_lock.test_and_set(std::memory_order::acquire)); // acquire lock
239 
240  job->m_next = nullptr; //clear pointer to successor
241  if (m_head == nullptr) { //if queue is empty

263  while (m_lock.test_and_set(std::memory_order::acquire)); // acquire lock
264 
265  JOB* head = m_head;
266  if (head != nullptr) { //if there is a job at the head of the queue
267  m_head = (JOB*)head->m_next; //let point m_head to its successor
268  m_size--; //decrease number of jobs
269  if (head == m_tail) { //if this is the only job

452  void thread_task(int32_t threadIndex = 0) noexcept {
453  constexpr uint32_t NOOP = 10; //number of empty loops until garbage collection
454  m_thread_index = threadIndex; //Remember your own thread index number
455  static std::atomic<uint32_t> thread_counter = m_thread_count.load(); //Counted down when started
456 
457  thread_counter--; //count down
458  while (thread_counter.load() > 0) {} //Continue only if all threads are running
459 
460  uint32_t next = rand() % m_thread_count; //initialize at random position for stealing
461  thread_local uint32_t noop = NOOP; //number of empty loops until threads sleeps
462  while (!m_terminate) { //Run until the job system is terminated
463  m_current_job = m_local_queues[m_thread_index].pop(); //try get a job from the local queue
464  if (m_current_job == nullptr) {
465  m_current_job = m_global_queues[m_thread_index].pop(); //try get a job from the global queue
466  }
467  if (m_current_job == nullptr) { //try steal job from another thread
468  if (++next >= m_thread_count) next = 0;
469  m_current_job = m_global_queues[next].pop();
470  }

473  std::chrono::high_resolution_clock::time_point t1, t2;
474 
475  if (is_logging()) {

477  }
478

480  auto type = m_current_job->m_type;
481  auto id = m_current_job->m_id;
482 
483  (*m_current_job)(); //if any job found execute it - a coro might be destroyed here!
484 
485  if (is_logging()) {
