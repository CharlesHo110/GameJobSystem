457  void thread_task(int32_t threadIndex = 0) noexcept {
458  constexpr uint32_t NOOP = 10; //number of empty loops until garbage collection
459  m_thread_index = threadIndex; //Remember your own thread index number
460  static std::atomic<uint32_t> thread_counter = m_thread_count.load(); //Counted down when started
461 
462  thread_counter--; //count down
463  while (thread_counter.load() > 0) {} //Continue only if all threads are running
464 
465  uint32_t next = rand() % m_thread_count; //initialize at random position for stealing
466  thread_local uint32_t noop = NOOP; //number of empty loops until threads sleeps
467  while (!m_terminate) { //Run until the job system is terminated
468  m_current_job = m_local_queues[m_thread_index].pop(); //try get a job from the local queue
